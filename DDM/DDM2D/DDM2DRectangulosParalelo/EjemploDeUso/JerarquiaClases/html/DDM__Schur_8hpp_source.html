65  inline void multiplica(Vector *u, Vector *v)
66  {
67  int i, n, k, j;
68  v->inicializa(0.0);
69  // Calculo del vector mediante la multiplicación de x por matriz de Schur Si de cada subdominio
70  for (i = 0; i < num_sub; i++)
71  {
72  for (n = 0, j = 0; j < num_nodos_frontera_interior; j++)
73  {
74  for (k = 0; k < num_nodos_elemento; k++)
75  {
76  if (subdominio[j][k] == i)
77  {
78  pXU[num_nodo_local[j][k]] = u->retorna(j);
79  n++;
80  }
81  }
82  }
83  funPar05(i, pXU, num_nodos_frontera_interior);
84 
85  for (j = 0; j < num_nodos_frontera_interior; j++)
86  {
87  for (k = 0; k < num_nodos_elemento; k++)
88  {
89  if (subdominio[j][k] == i) v->asigna(j,v->retorna(j) + pXU[num_nodo_local[j][k]]);
90  }
91  }
92  }
93  }
